session_stats <- map_dfr(session, function(sess) {
  data.frame(
    session_id = sess$date_exp[1],      
    mouse = sess$mouse_name[1],       
    n_neurons = nrow(sess$spks[[1]]),   
    n_trials = length(sess$feedback_type),  
    n_success = sum(sess$feedback_type == 1),  
    n_failure = sum(sess$feedback_type == -1), 
    success_rate = sum(sess$feedback_type == 1) / length(sess$feedback_type) 
  )
})

stimulus_stats <- map_dfr(session, function(sess) {
  sess_data <- data.frame(
    contrast_left = sess$contrast_left,
    contrast_right = sess$contrast_right
  )
  sess_data %>%
    group_by(contrast_left, contrast_right) %>%
    summarise(n = n(), .groups = "drop") %>%
    mutate(
      session_id = sess$date_exp[1],
      mouse = sess$mouse_name[1]
    )
})
