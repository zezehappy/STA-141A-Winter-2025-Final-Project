combined_data <- map_dfr(1:18, ~ {
  s <- sessions[[.x]]
  data.frame(
    session_id = .x,
    mouse = s$mouse_name,
    brain_area = s$brain_area,
    mean_left_contrast = mean(s$contrast_left),
    mean_right_contrast = mean(s$contrast_right),
    neuron_count = nrow(s$spks[[1]]),
    correct_feedback_ratio =mean(s$feedback_type == 1)
  )
})

p_success <- ggplot(combined_data, aes(x = mouse, y = correct_feedback_ratio, fill = mouse)) +
  geom_boxplot() +  
  stat_compare_means(method = "kruskal.test", label = "p.format", 
                     label.x = 1.5, label.y = max(combined_data$correct_feedback_ratio) * 1.05) + 
  labs(x = "Mouse", y = "Success Rate (Correct Feedback Ratio)", 
       title = "Success Rate Across Mice (Kruskal-Wallis Test)") +
  theme(
    panel.background = element_rect(fill = "grey90"),  
    panel.grid.major = element_line(color = "white"), 
    panel.grid.minor = element_line(color = "white"),  
    plot.title = element_text(hjust = 0.5)  
  )

print(p_success)

ggsave("success_rate_boxplot.png", plot = p_success, dpi = 500)
