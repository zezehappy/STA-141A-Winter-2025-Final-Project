extract_features <- function(session_id) {
  sess <- sessions[[session_id]]
  
  brain_areas <- unique(sess$brain_area)
  num_trials <- length(sess$contrast_left)
  time_windows <- 4  
  
  features <- matrix(0, nrow = num_trials, ncol = length(brain_areas) * time_windows)
  colnames(features) <- paste(rep(brain_areas, each = time_windows), 
                             rep(paste0("window", 1:time_windows), times = length(brain_areas)), 
                             sep = "_")
  
  for (trial in 1:num_trials) {
    spks <- sess$spks[[trial]]  
    window_size <- ncol(spks) %/% time_windows
    
    for (w in 1:time_windows) {
      start_bin <- (w - 1) * window_size + 1
      end_bin <- w * window_size

      for (area in brain_areas) {
        neurons_in_area <- which(sess$brain_area == area)
        if (length(neurons_in_area) > 0) {
          avg_activity <- mean(rowSums(spks[neurons_in_area, start_bin:end_bin, drop = FALSE]))
          col_name <- paste(area, paste0("window", w), sep = "_")
          features[trial, col_name] <- avg_activity
        }
      }
    }
  }
  return(features)
}
